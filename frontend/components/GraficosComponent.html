import { AfterViewInit, Component, Input } from '@angular/core';
import { AplicativoService } from '../../services/aplicativo.service';
import { ContaService } from '../../services/conta.service';

/**
 * Componente de Gráficos
 */
@Component({
  selector: 'app-graficos',
  templateUrl: './graficos.component.html',
  styleUrls: ['./graficos.component.scss']
})
export class GraficosComponent implements AfterViewInit {

  /**
   * identificador primário de conta
   */
  @Input() idConta?: string = "";
  
  /**
   * Definição de Dados vazia
   */
  doughnutChartData = {};  
  
  /**
   * Opções do Gráfico
   */
  options = {
    title: {
        display: false,
        text: '',
        fontSize: 16
    },
    legend: {
        position: 'none'
    }
  };  

  /**
   * Constructor
   * 
   * @param aplicativoService 
   * @param contaService 
   */
  constructor(public aplicativoService: AplicativoService,
              private contaService: ContaService) { }


  /**
   * Função de Inicio da Exibição
   */
  ngAfterViewInit(): void {

    this.contaService.getById(this.idConta || '').then(
      (a: any) => {        
        a.subscribe( (b: any) => {          
          if (b.success) {
            this.doughnutChartData = {
              labels: ['CREDITOS','DEBITOS','FUTURO'],
              datasets: [
                {
                  data: [b.saldoCredito, b.saldoDebito, b.saldoFuturo],              
                  backgroundColor: [                                    
                    "#36A2EB",
                    "#FF6384",
                    "#46AAEB",
                  ],
                  hoverBackgroundColor: [          
                    "#FAF54F",
                    "#FAF54F",
                    "#FAF54F"
                  ]
                }
              ]
            };            
          }
        });
      }
    )
  } 

}
