import { Component, OnInit } from '@angular/core';
import { ConfirmationService, MessageService } from 'primeng/api';
import { DialogService } from 'primeng/dynamicdialog';
import { ILancamento } from '../../interfaces/ilancamento.interface';
import { Lancamento } from '../../models/lancamento.model';
import { AplicativoService } from '../../services/aplicativo.service';
import { GraficosComponent } from '../graficos/graficos.component';
import { LancamentoDetalheComponent } from '../lancamento-detalhe/lancamento-detalhe.component';
import { LancamentosComponent } from '../lancamentos/lancamentos.component';
import { SplashScreenService } from 'src/_modules/core/services/splash-screen.service';
import { ModoDeEdicaoFormulario } from 'src/_modules/core/enums/modo-de-edicao-formulario';

/**
 * Componente Base do Aplicativo
 */
@Component({
  selector: 'app-aplicativo',
  templateUrl: './aplicativo.component.html',
  styleUrls: ['./aplicativo.component.scss']
})
export class AplicativoComponent implements OnInit {

  lancamento?: ILancamento;
  lancamentos: ILancamento[] = [];
  lancamentosSelecionados: ILancamento[] = [];

  sortOrder: any;
  sortField: any;
  

  /**
   * Constructor
   * 
   * @param aplicativoService 
   * @param splashScreenService 
   * @param dialogService 
   * @param messageService 
   * @param confirmationService 
   */
  constructor(public aplicativoService: AplicativoService,
              public splashScreenService: SplashScreenService,
              private dialogService: DialogService,
              private messageService: MessageService, 
              private confirmationService: ConfirmationService) { }

  /**
   * Função OnInit
   */
  ngOnInit(): void {
    //this.lancamentos = this.aplicativoService.financas.lancamentos;
    /*
    const utilizadores = this.firebaseService.getAllUtilizadores();    
    console.log("utilizadores", utilizadores);*/
  }

  /**
   * Consultar Lançamento
   * 
   * @param lancamento 
   */
  consultarLancamento(lancamento: ILancamento): void {
    const ref = this.dialogService.open(LancamentoDetalheComponent, {
      data: {
        id: lancamento.id,
        modo: ModoDeEdicaoFormulario.Consultar
      },
      closeOnEscape: false,
      header: 'DETALHE',
      width: '60%'
    });
  }

  /**
   * Incluir Lançamento
   */
  incluirLancamento(): void {
    this.lancamento = new Lancamento();

    const ref = this.dialogService.open(LancamentoDetalheComponent, {
      data: {
        id: 0,
        modo: ModoDeEdicaoFormulario.Incluir
      },
      closeOnEscape: true,
      showHeader: false,
      header: 'NOVO LANÇAMENTO',
      width: '60%',      
    });    

  }

  /**
   * Editar Lançamento
   * 
   * @param lancamento 
   */
  editarLancamento(lancamento: ILancamento): void {
    const ref = this.dialogService.open(LancamentoDetalheComponent, {
      data: {
        id: lancamento.id,
        modo: ModoDeEdicaoFormulario.Editar
      },
      closeOnEscape: false,
      header: 'EDITAR LANÇAMENTO',
      width: '60%'
    });
  }

  /**
   * Excluir Lançamento
   * 
   * @param lancamento 
   */
  excluirLancamento(lancamento: ILancamento): void {
    this.confirmationService.confirm({
      message: 'Are you sure you want to delete ' + lancamento.titulo + '?',
      header: 'Confirm',
      icon: 'pi pi-exclamation-triangle',
      accept: () => {
        this.lancamentos = this.lancamentos.filter(val => val.id !== lancamento.id);
        this.lancamento = new Lancamento();
  
        /*
        this.utilizadorService.excluirUtilizador(utilizador).subscribe((x) => {
          this.messageService.add({severity:'success', summary: 'Successful', detail: 'Utilizador Excluido', life: 3000});
        })
        */
      }
    });    
  }

  /**
   * Excluir lançamentos selecionados
   */
  excluirLancamentosSelecionados(): void {
    this.confirmationService.confirm({
      message: 'Are you sure you want to delete the selected users?',
      header: 'Confirm',
      icon: 'pi pi-exclamation-triangle',
      accept: () => {
        this.lancamentos = this.lancamentos.filter(val => !this.lancamentosSelecionados?.includes(val));
        this.lancamentosSelecionados?.forEach((data) => {
  
          /*
          this.utilizadorService.excluirUtilizador(data).subscribe((x) => {
            this.messageService.add({severity:'success', summary: 'Successful', detail: 'Utilizador Excluido', life: 3000});
          })
          */
          
        });  
        this.lancamentosSelecionados = [];
      }
    });
  }

  /**
   * Abrir Lançamentos
   */
  abrirLancamentos(): void {
  const ref = this.dialogService.open(LancamentosComponent, {
    data: {
      modo: null
    },
    closeOnEscape: false,
    header: 'LANCAMENTOS',
    width: '90%'    
  });
}

/**
 * Abrir Gráficos
 */
abrirGraficos(): void {
  const ref = this.dialogService.open(GraficosComponent, {
    data: {
      modo: null
    },
    closeOnEscape: false,
    header: 'GRAFICOS',
    width: '60%'    
  });
}


}
